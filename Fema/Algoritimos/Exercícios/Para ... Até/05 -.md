````
ALGORITIMO CINCO;
VAR
V[10],S,M: REAL;
I,CONT100,CONT50 : INT
INICIO 
PARA (I <- 0; I < 10; I++)
	IMPRIMA 'Digite um número';
	LEIA V[I];
	SE(V[I] < 100) 
	ENTÃO CONT100 <- CONT100 + 1;
	SE(V[I] < 50)
	ENTÃO S <- S + V[I]
		CONT50 <- CONT50 + 1;
}
SE(CONT100 == 0)
ENTÃO IMPRIMA 'NÃO FOI DIGITADO UM NÚMERO MAIOR QUE 100';
SENÃO IMPRIMA 'A QUANTIDADE É', CONT100;

SE(CONT50 == 0)
IMPRIMA 'FOI DIGITADO UM NÚMERO MAIOR QUE 50.';
ENTÃO 
M <- S / CONT50;
IMPRIMA 'A média de todos os números menor que 50', M;
FIM
